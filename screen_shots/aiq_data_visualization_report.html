# Analyze sales trends over time
merged_sale_user__weather_data['order_date'] = pd.to_datetime(merged_sale_user__weather_data['order_date'])
sales_trends_over_time = merged_sale_user__weather_data.groupby(merged_sale_user__weather_data['order_date'].dt.to_period('M'))['price'].sum()
# sales_trends_over_time.head()
plt.figure(figsize=(10, 6))
plt.plot(sales_trends_over_time.index.to_timestamp(), sales_trends_over_time.values, marker='o', linestyle='-')
plt.xlabel('Date')
plt.ylabel('Total Sales Amount')
plt.title('Sales Trends Over Time')
plt.xticks(rotation=45)
plt.grid(True)
plt.tight_layout()
plt.show()
